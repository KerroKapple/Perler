# Print-PDF Export Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add print-ready PDF export to BeadLab — replacing the current 4× PNG so paying users get an actual multi-page printable pattern with grid markers, color legend and bead counts.

**Architecture:** Pure spec builder (`buildPdfSpec`, already TDD'd) decoupled from a thin renderer (`drawPdf`) that calls jsPDF. UI wiring keeps the existing credits/paywall path; only the artifact changes. Pattern → grid array (palette indices) → spec → PDF blob.

**Tech Stack:** Vitest 2.x + jsdom · jsPDF 2.x (UMD via CDN in browser, npm in tests) · existing inline `<script>` becomes a thin caller of `src/lib.js` ESM module.

**Pre-existing (already done in prior commits):**
- `src/lib.js` exports `buildPdfSpec(pattern)` with 4 passing tests in `tests/buildPdfSpec.test.js`.
- Tests cover: single page small pattern, legend sort desc, ruler markers every 10 cells, multi-page split at 48-cell boundary.

function makeRulers(size) {
  const last = size - 1;
  const r = [];
  for (let i = 0; i < size; i += 10) {
    if (i !== last) r.push(i);
  }
  r.push(last);
  return r;
}

function hexToRgb(h) {
  const v = h.replace('#', '');
  return [
    parseInt(v.slice(0, 2), 16),
    parseInt(v.slice(2, 4), 16),
    parseInt(v.slice(4, 6), 16),
  ];
}

export function gridFromImageData(rgba, width, height, palette) {
  const pal = palette.map(hexToRgb);
  const out = new Array(width * height);
  for (let i = 0; i < width * height; i++) {
    const r = rgba[i * 4], g = rgba[i * 4 + 1], b = rgba[i * 4 + 2];
    let best = 0, bd = Infinity;
    for (let p = 0; p < pal.length; p++) {
      const dr = r - pal[p][0], dg = g - pal[p][1], db = b - pal[p][2];
      const d = dr * dr + dg * dg + db * db;
      if (d < bd) { bd = d; best = p; }
    }
    out[i] = best;
  }
  return out;
}
```

export function drawPdf(spec, JsPdfCtor) {
  const pdf = new JsPdfCtor({ unit: 'pt', format: spec.pageSize });
  spec.pages.forEach((page, i) => {
    if (i > 0) pdf.addPage();
    if (i === 0) {
      pdf.setFontSize(18);
      pdf.text(spec.title, 40, 50);
    }
    if (page.kind === 'grid') drawGridPage(pdf, page);
    else if (page.kind === 'legend') drawLegendPage(pdf, page);
  });
  return pdf.output('blob');
}

function drawGridPage(pdf, page) {
  // Minimal: just record dimensions; visual fidelity is V2.
  pdf.setFontSize(10);
  pdf.text(`Grid ${page.size.cols}×${page.size.rows}`, 40, 80);
}

function drawLegendPage(pdf, page) {
  pdf.setFontSize(10);
  page.entries.forEach((e, idx) => {
    pdf.text(`${e.code}  ${e.hex}  ×${e.count}`, 40, 60 + idx * 14);
  });
}
```

```js
function downloadPdf(){
  if(getCredits()<=0){showPaywall();return}
  setCredits(getCredits()-1);
  const cv = $('cv-bead');
  const w = cv.width, h = cv.height;
  const pal = PALETTES[$('palette').value].map(p => p[1]);
  const ctx = cv.getContext('2d');
  const data = ctx.getImageData(0, 0, w, h).data;
  const size = parseInt($('size').value);
  // Sample one pixel per bead-cell (cell side = w/size, take center)
  const cell = w / size;
  const sampled = new Uint8ClampedArray(size * size * 4);
  for(let y = 0; y < size; y++){
    for(let x = 0; x < size; x++){
      const sx = Math.floor(x * cell + cell / 2);
      const sy = Math.floor(y * cell + cell / 2);
      const si = (sy * w + sx) * 4;
      const di = (y * size + x) * 4;
      sampled[di] = data[si];
      sampled[di+1] = data[si+1];
      sampled[di+2] = data[si+2];
      sampled[di+3] = 255;
    }
  }
  const grid = gridFromImageData(sampled, size, size, pal);
  const spec = buildPdfSpec({
    size, grid, palette: pal,
    title: 'BeadLab Pattern · ' + new Date().toLocaleDateString(),
    author: 'You',
  });
  const blob = drawPdf(spec, window.jspdf.jsPDF);
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `beadlab_${Date.now()}.pdf`;
  a.click();
  setTimeout(()=>URL.revokeObjectURL(url), 1000);
  showToast(`PDF downloaded · ${getCredits()} credits left`);
}
```
